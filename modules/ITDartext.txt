(function() {
    'use strict';

    const style = document.createElement('style');
    style.innerHTML = `

        .drawing-modal.fullscreen-mode {
            width: 100vw !important;
            height: 100vh !important;
            max-width: none !important;
            max-height: none !important;
            top: 0 !important;
            left: 0 !important;
            border-radius: 0 !important;
            display: flex;
            flex-direction: column;
            z-index: 99999 !important;
        }

        .fullscreen-mode .drawing-canvas-container {
            flex-grow: 1 !important;
            height: auto !important;
            width: 100% !important;
            padding: 0 !important;
            background: #000 !important; 
        }

        .itd-fs-button-vibrant {
            background-color: #00ff41 !important; 
            color: #000000 !important;           
            border: none !important;
            padding: 6px 14px !important;
            margin-right: 15px !important;
            border-radius: 4px !important;
            cursor: pointer !important;
            font-size: 12px !important;
            font-weight: 800 !important;
            text-transform: uppercase !important;
            letter-spacing: 0.5px !important;
            transition: all 0.2s ease !important;
            box-shadow: 0 0 10px rgba(0, 255, 65, 0.3) !important;
        }

        .itd-fs-button-vibrant:hover {
            background-color: #00cc33 !important;
            box-shadow: 0 0 15px rgba(0, 255, 65, 0.6) !important;
            transform: translateY(-1px) !important;
        }

        .drawing-header {
            display: flex !important;
            align-items: center !important;
        }
    `;
    document.head.appendChild(style);

    function syncCanvasSize() {
        const modal = document.querySelector('.drawing-modal');
        const canvas = document.querySelector('canvas.drawing-canvas');
        const container = document.querySelector('.drawing-canvas-container');

        if (!modal || !canvas || !container) return;

        if (modal.classList.contains('fullscreen-mode')) {
            const rect = container.getBoundingClientRect();
            canvas.width = rect.width;
            canvas.height = rect.height;
        } else {
            canvas.width = 800;
            canvas.height = 267;
        }

        canvas.dispatchEvent(new Event('resize', { bubbles: true }));
    }

    function injectButton() {
        const header = document.querySelector('.drawing-header');
        if (header && !document.getElementById('itd-fs-btn')) {
            const btn = document.createElement('button');
            btn.id = 'itd-fs-btn';
            btn.className = 'itd-fs-button-vibrant';
            btn.innerText = 'РАСШИРИТЬ';

            const closeBtn = header.querySelector('.drawing-close');
            if (closeBtn) {
                header.insertBefore(btn, closeBtn);
            } else {
                header.appendChild(btn);
            }

            btn.onclick = (e) => {
                e.preventDefault();
                const modal = document.querySelector('.drawing-modal');
                if (modal) {
                    modal.classList.toggle('fullscreen-mode');
                    setTimeout(syncCanvasSize, 50);
                    btn.innerText = modal.classList.contains('fullscreen-mode') ? 'СВЕРНУТЬ' : 'РАСШИРИТЬ';
                }
            };
        }
    }

    const observer = new MutationObserver(() => injectButton());
    observer.observe(document.body, { childList: true, subtree: true });

    window.addEventListener('resize', () => {
        if (document.querySelector('.fullscreen-mode')) syncCanvasSize();
    });

})();
